/**
* We are both programmers,
* so I write some code to describe our stories.
*/
Boy i = new Boy("杨家印");
Girl u = new Girl("朱诗婕");
// Feb 4, 2019, I told you I love you.
i.love(u);
// Luckily, you accepted and became my girlfriend eversince.
u.accepted();
// Since then, I miss u every day.
i.miss(u);
// You said that your parents may not agree.
// So I keep waiting and I have confidence that you will be with me.
boolean isHesitate = true;
while (isHesitate) {
i.waitFor(u);
// I think it is an important decision
// and you should think it over.
isHesitate = u.thinkOver();
}
// After deep understanding, we are together.
i.pet(u);
u.pet(i);
